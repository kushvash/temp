class Solution {
public:
    int squaresCal(int n) {
        int ans=0, p;

        while(n){
            p=n%10;
            ans+=p*p;
            n/=10;
        }

        return ans;
    }

    bool isHappy(int n) {
        int s=n, f=n;

        while(s!=1 && f!=1){
            s=squaresCal(s);
            f=squaresCal(squaresCal(f));

            if(s==f){
                return false;
            }
        }

        return true;

    }
};
